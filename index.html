        const photo = document.getElementById("photo");

            const scrollY = window.scrollY;

            const windowHeight = window.innerHeight;


            if (scrollY >= windowHeight) {

                photo.classList.add("sticky");

                name.classList.add("sticky");

            } else {

                name.classList.remove("sticky");

                textBox1.classList.add("hidden");

                textBox1.classList.remove("visible");

            }


            if (scrollY >= windowHeight * 2) {

                experience1.classList.remove("hidden");

                experience1.classList.add("visible");

            } else {

                experience1.classList.add("hidden");

                experience1.classList.remove("visible");

            }


            if (scrollY >= windowHeight * 3) {

                experience2.classList.remove("hidden");

            } else {

                experience2.classList.add("hidden");

                experience2.classList.remove("visible");

            }
